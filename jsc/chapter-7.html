১। 3x - 75x3

সমাধান :

3x - 75x3

= 3x(1- 25x2 )

= 3x{(1) - (5x)2}

= 3x(1+ 5x)(1- 5x)


২। 4x2 - y2

সমাধানঃ

            4x2 - y2

                =(2x)2-y2

            =(2x+y)(2x-y)

           

৩। 3ay2 - 48a

সমাধানঃ

3ay2 - 48a

            =3a(y2 – 16)

            =3a(y2-42)

            =3a(y-4)(y+4)

 

৪। a2 - 2ab + b2 - p2  

সমাধানঃ

a2 - 2ab + b2 - p2

                = (a -  b)2 - p2

            =(a-b+p)(a-b-p)

           

৫। 16y2 - a2 - 6a - 9

সমাধানঃ

16y2 - a2 - 6a – 9

            =(4y)2 –( a2 + 2.a.3 – 32)

            =(4y)2-(a+3)2

            ={(4y+(a+3)}{(4y-(a+3)}

            =(4y+a+3)(4y-a-3)

 

৬। 8a + ap3

সমাধানঃ

8a + ap3

            =a(8 + p3)

            =a(23+pp3)

            =a(2+p)(22+2.p+p2)

            =a(2+pp)(4+2p+pp2)

 

৭। 2a3 +16b3  

সমাধানঃ

2a3 +16b3       

            =2(a3 +8b3)

            =2{a3+(2b)3}

            =2(a+2b){(a)2+a.2b+(2b)2}

            =2(a+2b)(a2+2ab+4b2)

           

 ৮। x2 + y2 - 2xy -1     

সমাধানঃ

x2 + y2 - 2xy -1

            = x2 - 2xy+ y2 -1

            =(x-y)2-12

            =(x-y+1)(x-y-1)

 

৯। a2 - 2ab + 2b -1

সমাধানঃ

a2 - 2ab + 2b -1

            = (a2 -1)- 2b(a -1)

            =(a-1)(a+1)-2b(a-1)

            =(a-1)(a+1-2b)

 

১০। x4 - 6x2 +1           

সমাধানঃ

x4 - 6x2 +1      

            =(x2 )2 – 2.x2 .1+12-4x2

            =(x2-1)2-(2x)2

                =(x2-1+2x)(x2-1-2x)

           

১১। 36 -12x + x2         

সমাধানঃ

36 -12x + x2

                =x2-2.x.6+62

            =(x-6)2

                =(x-6)(x+6)

 

১২। x6 - y6

সমাধানঃ

x6 - y6

=(x3)2-(y3)2

=(x3+Y3)( x3-Y3)

=(X+Y)(X2-XY+Y2)(X-Y)(X2+XY-Y2)

=(X+Y) (X-Y)( (X2-XY+Y2) X2+XY-Y2)

 

১৩। (x - y)3 + z3

সমাধানঃ

(x - y)3 + z3

            ={(x - y) + z}{(x-y)2-(x-y)z+z2}

            =(x-y+z)(x2-2xy+y2-xz+yz+z2)

           

১৪। 64x3 -8y3

সমাধানঃ

64x3 -8y3

=8(8x3 -y3)

=8{(2x)3-y3)}

=8(2x-y){(2x)2+2.x.y+(y)2}

=8(2x-y)(4x2+2xy+y2)

 

১৫। x2 +14x + 40

সমাধানঃ

x2 +14x + 40

            = x2 +4x+10x + 40

            =x(x+4)+10(x+4)

            =(x+4)(x+10)

 

১৬। x2 + 7x -120        

সমাধানঃ

x2 + 7x -120

            = x2 + 15x-8x -120

            =x(x+15)-8(x+15)

            =(x-8)(x+15)

 

১৭। x2 - 51x + 650

সমাধানঃ

x2 - 51x + 650

            = x2 - 26x-25x + 650

            =x(x-26)-25(x-26)

            =(x-25)(x-26)

 

১৮। a2 + 7ab +12b2

সমাধানঃ

a2 + 7ab +12b2

            = a2 + 3ab+4ab +12b2

            =a(a+3b)+4b(a+3b)

            =(a+4b)(a+3b)

 

১৯। p2 + 2 pq -80b2  

সমাধানঃ

p2 + 2 pq -80b2          

            = p2 + 10pq-8pq -80b2         

            =p(p+10q)-8q(p+10q)

            =(p-8q)(p+10q)

 

২০। x2 - 3xy - 40y2     

সমাধানঃ

x2 - 3xy - 40y2

            = x2 - 8xy+5xy - 40y2

            =x(x-8y)+5y(x-8y)

            =(x+5y)(x-8y)

 

২১। (x2 - x)2 + 3(x2 - x) – 40

সমাধানঃ

মনে করি, x2 – x=a

প্রাপ্ত রাশি=a2+3a-40

            = a2+8a-5a-40

            =a(a+8)-5(a+8)

            =(a-5)(a+8)

            =(x2-x-5)(x2-x+8) [মান বসিয়ে]


২২। (a2 + b2 )2 -18(a2 + b2 ) -88

সমাধানঃ

ধরি,  a2 + b2=x

প্রদত্ত রাশি=x2-18x-88

            = x2-22x+4x-88

            =x(x-22)+4(x-22)

            =(x+4)(x-22)

            =( a2 + b2+4)( a2 + b2-22)

 

২৩। (a2 + 7a)2 -8(a2 + 7a) -180

সমাধানঃ

মনে করি, a2 + 7a=x

প্রদত্ত রাশি=x2-8x-180

            = x2-18x+10x-180

            =x(x-18)+10(x-18)

            =(x+10)(x-18)

            =( a2 + 7a+10)( a2 + 7a-18)

 

২৪। x2 + (3a + 4b)x + (2a2 + 5ab + 3b2 )

সমাধানঃ

x2 + (3a + 4b)x + (2a2 + 5ab + 3b2 )

            = x2 + (3a + 4b)x + (2a2 + 3ab+2ab + 3b2 )

            = x2 + (3a + 4b)x +{a(2a+3b)+b(2a+3b)

            = x2 + (3a + 4b)x +(a+b)(2a+3b)

            = x2 + (2a + 3b)x+(a+b)x +(a+b)(2a+3b)

            =x(x+2a+3b)+(a+b)(x+2a+3b)

            =(x+a+b)(x+2a+3b)

 

২৫। 6x2 - x -15

সমাধানঃ

6x2 - x -15

            =6x2 - 10x+9x -15

            =2x(3x-5)+3(3x-5)

            =(2x+3)(3x-5)

 

২৬। x2 - x - (a +1)(a + 2)

সমাধানঃ

x2 - x - (a +1)(a + 2)

            = x2 - x - (a +1)(a + 1+1)

            = x2 - x - m(m+1)  [a+1=m ধরে]

            =x2-x-m2-m

            = x2-m2-x-m

            =(x-m)(x+m)-1(x+m)

            =(x+m)(x-m-1)

            =(x+a+1)(x-a-1-1) [মান বসিয়ে]

            =(x+a+1)(x-a-2)

 

২৭। 3x2 +11x – 4

সমাধানঃ

3x2 +11x – 4

            =3x2 +12x-x – 4

            =3x(x+4)-1(x+4)

            =(3x-1)(x+4)

 

২৮। 3x2 -16x -12

২৯। 2x2 - 9x - 35

৩০। 2x2 - 5xy + 2y2

৩১। x3 - 8(x - y)3

x3 - 8(x - y)3

৩২। 10p2 +11pq - 6q2

৩৩। 2(x + y)2 - 3(x + y) – 2

ধরি, x+y=a

প্রদত্ত রাশি=2(a)2 - 3(a) – 2

            =(2a+1)(a-2)

            =(2x+2y+1)(x+y-2)

 

৩৪। ax2 + (a2 +1)x + a

ax2 + (a2 +1)x + a

            =(ax+1)(x+1)

 

৩৫। 15x2 -11xy -12y2

15x2 -11xy -12y2

৩৬। a3 - 3a2b + 3ab2 - 2b3
